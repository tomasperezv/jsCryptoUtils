  7 /**
  8  * @class base64
  9  */
 10 function Base64() {
 11 
 12 	Algorithm.call();
 13 
 14 	/**
 15 	 * @var {String} base64Alpha
 16 	 * @visibility {Private}
 17 	 */
 18 	this.base64Alpha = "abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789& ";
 19 	
 20 	/**
 21 	 * @var encodeBlocksSize
 22 	 */
 23 	this.encodeBlocksSize = 3;
 24 	
 25 	/**
 26 	 * @var decodeBlocksSize
 27 	 */
 28 	this.decodeBlocksSize = 4;
 29 	
 30 	/**
 31 	 * Returns an string from an array with positions in the alpha string.
 32 	 * e.g. positions (1, 3); alpha = "abcd" => "bd"
 33 	 *
 34 	 * @author tom@0x101.com 
 35 	 * @visibility {Private}
 36 	 * @param {Array} posArray
 37 	 * @return {String}
 38 	 */
 39 	this.getAlphaSubStr = function (posArray) {
 40 		output = '';
 41 		for(var i in posArray) {
 42 			output += this.base64Alpha[posArray[i]];
 43 		}
 44 		return output;
 45 	};
 46 }

 48 Base64.prototype = new Algorithm();

 50 /**
 51  * Encode a text to base64.
 52  *
 53  * @author tom@0x101.com 
 54  * @visibility {Public}
 55  * @param {String} inputText
 56  * @return {String}
 57  */
 58 Base64.prototype.encode = function (inputText) {
 59 	var output = '';
 60 	if (typeof(inputText) === 'string') {
 61 		// fix the string length
 62 		var input = CryptoUtils.fixStringLength(inputText, this.encodeBlocksSize);
 63 		// init the stringBlocksIterator
 64 		var stringBlocksIterator = new StringBlocksIterator();
 65 		stringBlocksIterator.init(input, this.encodeBlocksSize);
 66 		
 67 		var subdata = null;
 68 		while ((subdata = stringBlocksIterator.current())) {
 69 			var x = subdata.charCodeAt(0);
 70 			var y = subdata.charCodeAt(1);
 71 			var z = subdata.charCodeAt(2);
 72 			
 73 			CryptoConsole.logArrayAsBin([x, y, z], 'base64.encode.input');
 74 			this.sendCallback(this.onBeforeIteration);
 75 			
 76 			// Apply the necessary bitwise operators
 77 			var _x = x >> 2;
 78 			var _y = ((x & 3) << 4) | (y >> 4);   
 79 			var _z = ((y & 15) << 2) | (z >> 6);
 80 			var _k = z & 63; 
 81 			
 82 			CryptoConsole.logArrayAsBin([_x, _y, _z, _k], 'base64.decode.output');
 83 			this.sendCallback(this.onAfterIteration);
 84 			
 85 			output += this.getAlphaSubStr([_x, _y, _z, _k]);
 86 			
 87 			// Process next block 
 88 			stringBlocksIterator.next();
 89 		}
 90 	}
 91 	return output;
 92 };

 94 /**
 95  * Decode a text from base64.
 96  *
 97  * @author tom@0x101.com 
 98  * @visibility {Public}
 99  * @param {String} inputText
100  * @return {String}
101  */
102 Base64.prototype.decode = function (inputText) {
103 	// TODO: fix special chars like áéíóú
104 	var output = '';
105 	if (typeof(inputText) === 'string') {
106 		// fix the string length
107 		input = CryptoUtils.fixStringLength(inputText, this.decodeBlocksSize);
108 		// init the stringBlocksIterator
109 		stringBlocksIterator = new StringBlocksIterator();
110 		stringBlocksIterator.init(input, this.decodeBlocksSize);
111 		
112 		var subdata = null;
113 		while((subdata = stringBlocksIterator.current())) {
114 			var x = this.base64Alpha.indexOf(subdata[0]);
115 			var y = this.base64Alpha.indexOf(subdata[1]);
116 			var z = this.base64Alpha.indexOf(subdata[2]);
117 			var k = this.base64Alpha.indexOf(subdata[3]);
118 			
119 			CryptoConsole.logArrayAsBin(new Array(x, y, z, k), 'base64.decode.input');
120 			this.sendCallback(this.onBeforeIteration);
121 			
122 			// Apply the bitwise operator to decode the base64 data
123 			var _x = ((x << 2)) | (y >> 4);
124 			var _y = ((y << 4) | (z >> 2));
125 			var _z = ((z << 6)) | k;
126 			
127 			// TODO: Review the following procedure to clean extra-bits
128 			// probably we could use a better type-conversion
129 			
130 			// Clean extra-bits
131 			_x = _x << 24;
132 			_x = _x >> 24;
133 			
134 			_y = _y << 24;
135 			_y = _y >> 24;
136 			
137 			_z = _z << 24;
138 			_z = _z >> 24;
139 			
140 			CryptoConsole.logArrayAsBin(new Array(_x, _y, _z), 'base64.decode.output');
141 			this.sendCallback(this.onAfterIteration);
142 			
143 			// Prepare the output
144 			output += String.fromCharCode(_x, _y, _z);
145 			
146 			// Process next data block
147 			stringBlocksIterator.next();
148 		}
149 	}
150 	return CryptoUtils.trim(output);
151 };
