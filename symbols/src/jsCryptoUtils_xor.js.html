 10 function Xor() {
 11 	Algorithm.call();
 12 }
 13 
 14 Xor.prototype = new Algorithm();
 15 
 16 /**
 17  * Apply a basic XOR encoding to a string. 
 18  * 
 19  * @author tom@0x101.com 
 20  * @visibility {Public}
 21  * @param {String} inputText
 22  * @param {String} key
 23  * @return {String}
 24  */    
 25 Xor.prototype.encode = function (inputText, key) {
 26 
 27 	var outputText = '';
 28 
 29 	if (typeof(inputText) === 'string') {
 30 		// init the stringBlocksIterator
 31 		inputIterator = new StringBlocksIterator();
 32 		inputIterator.init(inputText, this.encodeBlocksSize);
 33 		
 34 		keyIterator = new StringBlocksIterator();
 35 		keyIterator.init(key, this.encodeBlocksSize);
 36 		
 37 		var subdata = null;
 38 		while ((subdata = inputIterator.current())) {
 39 			var x = subdata.charCodeAt(0);
 40 			
 41 			// If we are at the end of the key, then move to the
 42 			// begin.
 43 			if(keyIterator.current() === null) {
 44 				keyIterator.reset();
 45 			}
 46 			
 47 			var y = keyIterator.current().charCodeAt(0);
 48 			
 49 			// Apply the xor operation.
 50 			var result = x ^ y;
 51 			
 52 			// Store the result in the output string
 53 			outputText += String.fromCharCode(result);
 54 			
 55 			CryptoConsole.logArrayAsBin([x, y, result], 'input, output, result');
 56 			this.sendCallback(this.onAfterIteration);
 57 			
 58 			inputIterator.next();
 59 			keyIterator.next();
 60 		}
 61     }
 62     
 63     return outputText;
 64 };
